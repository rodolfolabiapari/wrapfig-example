\usepackage{graphicx}

\title{Package Example: wrapfig}

\begin{document}

\centering
\includegraphics[width=0.25\textwidth]{frog.jpg}
\caption{\label{fig:frog1}This is a figure caption.}
\end{wrapfigure}

\lipsum[1]

\lipsum[2-3]
